		{
		x =0;
			
			
			if (this.rbtReino.Checked)

				x += 2;
				
			}
			if (this.rbtMuss.Checked)

			}

				x +=2;
				
			}
			if (this.monC.SelectionStart.Date.ToShortDateString()=="16/09/1810")
			{
				x +=2;
				
			}
		System.Windows.Forms.MessageBox.Show(x.ToString());	
		}
